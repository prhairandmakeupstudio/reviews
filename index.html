let selectedRating = 0; // Handle star click event document.querySelectorAll('.star').forEach(star => { star.addEventListener('click', function() { selectedRating = parseInt(star.getAttribute('data-value')); updateStars(selectedRating); // Automatically redirect when 5 stars are selected if (selectedRating === 5) { // Open the link in Chrome (browser-specific behavior may differ) window.location.href = 'https://g.page/r/CSEDTwJF5uMuEBM/review'; // Redirect to the review page } }); }); // Update star colors based on selected rating function updateStars(rating) { const stars = document.querySelectorAll('.star'); stars.forEach(star => { const starValue = parseInt(star.getAttribute('data-value')); if (starValue <= rating) { star.classList.add('selected'); } else { star.classList.remove('selected'); } }); // Display message for non-5-star ratings const responseMessage = document.getElementById('responseMessage'); if (rating < 5) { responseMessage.style.display = 'block'; responseMessage.textContent = 'Thank you for your review!'; } else { responseMessage.style.display = 'none'; } }
